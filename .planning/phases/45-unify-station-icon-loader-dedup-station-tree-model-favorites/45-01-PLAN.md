      ```python
      def load_station_icon(station: Station, size: int = 32) -> QIcon:
          """QPixmapCache-backed station icon resolved via abs_art_path with fallback.

          Single source of truth (D-01, D-02). Cache key uses the resolved
          absolute path (or FALLBACK_ICON resource path) per D-03.
          """
          abs_path = abs_art_path(station.station_art_path)
          load_path = abs_path or FALLBACK_ICON
          key = f"station-logo:{load_path}"
          pix = QPixmap()
          if not QPixmapCache.find(key, pix):
              pix = QPixmap(load_path)
              if pix.isNull():
                  pix = QPixmap(FALLBACK_ICON)
              pix = pix.scaled(size, size, Qt.KeepAspectRatio, Qt.SmoothTransformation)
              QPixmapCache.insert(key, pix)
          return QIcon(pix)
      ```
