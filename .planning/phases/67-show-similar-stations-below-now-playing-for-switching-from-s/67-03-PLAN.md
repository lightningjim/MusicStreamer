def _on_sibling_link_activated(self, href: str) -> None:
    prefix = "sibling://"
    if not href.startswith(prefix): return
    try: sibling_id = int(href[len(prefix):])
    except ValueError: return
    if self._station is None or self._station.id == sibling_id: return
    try: sibling = self._repo.get_station(sibling_id)
    except Exception: return
    if sibling is None: return
    self.sibling_activated.emit(sibling)
