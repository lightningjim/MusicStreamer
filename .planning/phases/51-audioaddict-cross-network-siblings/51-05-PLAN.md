    ```python
    def test_phase_51_sibling_navigation_end_to_end(
        qtbot, fake_player, fake_repo, monkeypatch
    ):
        """Phase 51 SC #1, #2, #3, #4 end-to-end:

        - SC #1: DI.fm Ambient dialog renders 'Also on:' with ZenRadio link
        - SC #2: Clicking the link opens EditStationDialog for the ZenRadio sibling
        - SC #3: No aa_channel_key column on Station — siblings derived from URL
        - SC #4: Cross-network failover NOT introduced — fake_player.play never called
        """
        from PySide6.QtWidgets import QDialog
        from unittest.mock import MagicMock
        from musicstreamer.models import Station, StationStream, Provider
        from musicstreamer.ui_qt import edit_station_dialog as esd_mod
        from musicstreamer.ui_qt.main_window import MainWindow

        # ── ARRANGE ────────────────────────────────────────────────────────
        di_station = Station(
            id=1, name="Ambient", provider_id=1, provider_name="DI.fm",
            tags="", station_art_path=None, album_fallback_path=None,
            icy_disabled=False,
            streams=[StationStream(
                id=10, station_id=1,
                url="http://prem1.di.fm:80/ambient_hi?listen_key=abc",
                position=1,
            )],
        )
        zen_station = Station(
            id=2, name="Ambient", provider_id=2, provider_name="ZenRadio",
            tags="", station_art_path=None, album_fallback_path=None,
            icy_disabled=False,
            streams=[StationStream(
                id=20, station_id=2,
                url="http://prem1.zenradio.com/zrambient?listen_key=abc",
                position=1,
            )],
        )

        # Configure fake_repo to return both stations from list_stations() and
        # the appropriate one from get_station(). Adjust attribute names if
        # fake_repo is a real fixture rather than MagicMock — see file head.
        fake_repo.list_stations = MagicMock(return_value=[di_station, zen_station])
        fake_repo.get_station = MagicMock(side_effect=lambda sid: {1: di_station, 2: zen_station}.get(sid))
        fake_repo.list_streams = MagicMock(side_effect=lambda sid: {1: di_station.streams, 2: zen_station.streams}.get(sid, []))
        fake_repo.list_providers = MagicMock(return_value=[Provider(1, "DI.fm"), Provider(2, "ZenRadio")])
        fake_repo.ensure_provider = MagicMock(return_value=1)

        # ── MONKEYPATCH EditStationDialog.exec ─────────────────────────────
        exec_calls: list[int] = []
        captured_sibling_label_text: list[str] = []

        def _fake_exec(self):
            exec_calls.append(self._station.id)
            if len(exec_calls) == 1:
                # First exec: this is the DI.fm dialog. Capture sibling-label
                # state and simulate a link click.
                captured_sibling_label_text.append(self._sibling_label.text())
                # Programmatically trigger the slot the linkActivated signal
                # would normally fire.
                self._on_sibling_link_activated("sibling://2")
            return QDialog.Accepted

        monkeypatch.setattr(esd_mod.EditStationDialog, "exec", _fake_exec)

        # ── ACT ────────────────────────────────────────────────────────────
        w = MainWindow(fake_player, fake_repo)
        qtbot.addWidget(w)
        w._on_edit_requested(di_station)

        # ── ASSERT ─────────────────────────────────────────────────────────
        # SC #2: clicking the link opened the sibling's edit dialog
        assert exec_calls == [1, 2], f"expected [1, 2], got {exec_calls}"

        # SC #1: 'Also on:' rendered with a sibling://2 link to ZenRadio
        assert len(captured_sibling_label_text) == 1
        text = captured_sibling_label_text[0]
        assert "Also on:" in text, f"sibling label missing 'Also on:': {text!r}"
        assert 'href="sibling://2"' in text, f"sibling label missing href: {text!r}"
        assert "ZenRadio" in text, f"sibling label missing ZenRadio: {text!r}"

        # SC #4: no playback method called during sibling navigation.
        # fake_player is a MagicMock-based fixture; if .play was never called,
        # called=False. (If fake_player is a real class, swap for the appropriate
        # spy mechanism — see the file's other tests for precedent.)
        if hasattr(fake_player, "play") and hasattr(fake_player.play, "called"):
            assert fake_player.play.called is False, "SC #4 violation: player.play called during sibling navigation"
        if hasattr(fake_player, "failover") and hasattr(fake_player.failover, "called"):
            assert fake_player.failover.called is False, "SC #4 violation: player.failover called during sibling navigation"

        # SC #3: Station dataclass has NO aa_channel_key field — sibling detection
        # is purely URL-derived (D-01). Verify by inspecting the dataclass fields.
        from dataclasses import fields as dc_fields
        station_field_names = {f.name for f in dc_fields(Station)}
        assert "aa_channel_key" not in station_field_names, \
            "SC #3 violation: Station gained aa_channel_key field — D-01 says no schema change"
    ```
