```python
def fetch_mb_cover(artist: str, title: str, callback) -> None:
    """Fetch MB+CAA cover for an artist/title pair (D-07..D-20).

    The caller is responsible for the ' - ' split (D-07) — this function
    receives already-split parts. Bare-title cases never reach here.

    On hit: writes musicstreamer.cover_art.last_itunes_result = {
        'artwork_url': '<caa_url>', 'genre': '<mb_tag_or_empty>'
    } and invokes callback(temp_jpg_path).

    On any miss (no recording, no release, CAA 404, network error, HTTP 503,
    JSON parse error, anything): callback(None). Never raises.

    Submits via _pending (queue.Queue maxsize=1) — older queued submissions
    are dropped to honor latest-wins per D-13.
    """
```
