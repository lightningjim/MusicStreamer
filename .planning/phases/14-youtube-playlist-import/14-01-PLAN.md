def station_exists_by_url(self, url: str) -> bool:
    row = self.con.execute(
        "SELECT 1 FROM stations WHERE url = ?", (url,)
    ).fetchone()
    return row is not None

def insert_station(self, name: str, url: str, provider_name: str, tags: str) -> int:
    provider_id = self.ensure_provider(provider_name) if provider_name else None
    cur = self.con.execute(
        "INSERT INTO stations(name, url, provider_id, tags) VALUES (?, ?, ?, ?)",
        (name, url, provider_id, tags or ""),
    )
    self.con.commit()
    return int(cur.lastrowid)
```

import re
def is_yt_playlist_url(url: str) -> bool:
    return bool(
        re.search(r"youtube\.com/playlist\?.*list=", url) or
        re.search(r"youtube\.com/@[^/]+/(streams|live|videos)", url)
    )
```

2. `scan_playlist(url: str) -> list[dict]` — Runs yt-dlp subprocess, parses JSON lines, filters is_live:
```python
import subprocess, json

def scan_playlist(url: str) -> list[dict]:
    result = subprocess.run(
        ["yt-dlp", "--flat-playlist", "--dump-json", url],
        capture_output=True, text=True, timeout=120,
    )
    if result.returncode != 0:
        stderr = result.stderr.strip()
        if "private" in stderr.lower() or "unavailable" in stderr.lower():
            raise ValueError("Playlist Not Accessible")
        raise RuntimeError(stderr or "yt-dlp failed")
    entries = []
    for line in result.stdout.splitlines():
        line = line.strip()
        if not line:
            continue
        try:
            entry = json.loads(line)
        except json.JSONDecodeError:
            continue
        if entry.get("is_live") is True:
            entries.append({
                "title": entry.get("title", "Untitled"),
                "url": entry.get("url") or entry.get("webpage_url"),
                "provider": entry.get("playlist_channel") or entry.get("playlist_uploader", ""),
            })
    return entries
```

3. `import_stations(entries: list[dict], repo, on_progress=None) -> tuple[int, int]` — Import loop with progress callback:
```python
def import_stations(entries, repo, on_progress=None):
    imported = 0
    skipped = 0
    for entry in entries:
        url = entry["url"]
        if repo.station_exists_by_url(url):
            skipped += 1
        else:
            repo.insert_station(
                name=entry["title"],
                url=url,
                provider_name=entry["provider"],
                tags="",
            )
            imported += 1
        if on_progress:
            on_progress(imported, skipped)
    return imported, skipped
```
