    ```python
    def _start_glitch_ramp(self, direction: str) -> None:
        """Phase 57 / WIN-03 glitch fix: start a QTimer-driven volume ramp.

        direction = "down" (pre-NULL fade to 0) or "up" (post-PLAYING fade to cached _volume).
        Mirrors Phase 52 _start_eq_ramp (lines 746-786): main-thread QTimer,
        ramp-state dict, reverse-from-current on mid-ramp interruption.

        Composes with Plan 57-03's volume fix without double-writing — this
        method writes to the target Plan 57-03 chose:
        - Option A / hybrid: playbin3.volume
        - Option B: self._volume_element
        """
        # Snapshot the current write-target's value so reverse-mid-ramp re-captures
        # live state (Phase 52 D-05 idiom).
        if self._volume_element is not None:
            current = self._volume_element.get_property("volume")
        else:
            current = self._pipeline.get_property("volume")

        target = 0.0 if direction == "down" else self._volume

        # Fresh ramp; cancel any in-flight ramp first (Pitfall: dangling state).
        self._glitch_ramp_timer.stop()

        self._glitch_ramp_state = {
            "direction": direction,
            "from": current,
            "to": target,
            "tick": 0,
            "ticks": self._GLITCH_RAMP_TICKS,
        }
        self._glitch_ramp_timer.start()

    def _on_glitch_ramp_tick(self) -> None:
        """Phase 57 / WIN-03 glitch fix: per-tick interpolated volume write.

        Mirrors Phase 52 _on_eq_ramp_tick (lines 788-813)."""
        state = self._glitch_ramp_state
        if state is None:
            self._glitch_ramp_timer.stop()
            return

        state["tick"] += 1
        progress = min(1.0, state["tick"] / state["ticks"])
        value = state["from"] + (state["to"] - state["from"]) * progress
        # Clamp to [0.0, 1.0] for safety (intermediate float arithmetic).
        value = max(0.0, min(1.0, value))

        # Write to the Plan-57-03 target. NEVER both — composition rule.
        if self._volume_element is not None:
            self._volume_element.set_property("volume", value)
        else:
            self._pipeline.set_property("volume", value)

        if state["tick"] >= state["ticks"]:
            self._glitch_ramp_timer.stop()
            self._glitch_ramp_state = None
    ```

    **Step 5: Add the synchronous ramp helper** (`_glitch_ramp_now`):

    ```python
    def _glitch_ramp_now(self, direction: str) -> None:
        """Phase 57 / WIN-03 glitch fix: drive the ramp to completion synchronously.

        On the main thread, processEvents() between ticks lets QTimer fire
        all 8 ticks within the 40 ms window before the caller proceeds. This
        is the "pause()" entry point — async wouldn't compose with the
        immediate set_state(NULL) that follows.

        For the post-PLAYING fade-up (in play() and _set_uri), use the
        async variant (_start_glitch_ramp) — the user already hears audio
        starting, so a 40 ms ramp-up over the start is imperceptible.
        """
        from PySide6.QtCore import QCoreApplication, QEventLoop

        # Snapshot current target value
        if self._volume_element is not None:
            current = self._volume_element.get_property("volume")
        else:
            current = self._pipeline.get_property("volume")

        target = 0.0 if direction == "down" else self._volume
        ticks = self._GLITCH_RAMP_TICKS

        for tick in range(1, ticks + 1):
            progress = tick / ticks
            value = current + (target - current) * progress
            value = max(0.0, min(1.0, value))
            if self._volume_element is not None:
                self._volume_element.set_property("volume", value)
            else:
                self._pipeline.set_property("volume", value)
            # Yield to the event loop briefly (~5 ms per tick); flushes
            # GStreamer property writes through to the sink before the
            # next iteration.
            # WARNING 1 fix: ExcludeUserInputEvents prevents reentrancy of
            # slider drags / button clicks / other timers into pause /
            # play / set_volume / _set_uri during the 40 ms fade-down
            # inside pause(). User-input events queue rather than re-enter;
            # GStreamer property writes still flush.
            QCoreApplication.processEvents(
                QEventLoop.ExcludeUserInputEvents, self._GLITCH_RAMP_INTERVAL_MS
            )
    ```
