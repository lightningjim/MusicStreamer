  Template:
  ```python
  """Phase 36 plan 04 — PORT-07 Windows Fusion palette code-path test.

  The `_apply_windows_palette` helper is exercised directly under offscreen Qt.
  We mock QGuiApplication.styleHints().colorScheme() to drive the Dark vs Light
  branches without needing a real Windows VM.
  """
  from unittest.mock import patch, MagicMock

  from PySide6.QtCore import Qt
  from PySide6.QtGui import QColor, QPalette

  from musicstreamer.__main__ import _apply_windows_palette


  def _mock_style_hints(scheme):
      hints = MagicMock()
      hints.colorScheme.return_value = scheme
      return hints


  def test_apply_windows_palette_dark(qapp):
      with patch(
          "musicstreamer.__main__.QGuiApplication.styleHints",
          return_value=_mock_style_hints(Qt.ColorScheme.Dark),
      ):
          _apply_windows_palette(qapp)
      window_color = qapp.palette().color(QPalette.ColorRole.Window)
      assert window_color == QColor(53, 53, 53)


  def test_apply_windows_palette_light_noop(qapp):
      # Snapshot pre-call window color
      pre = qapp.palette().color(QPalette.ColorRole.Window)
      with patch(
          "musicstreamer.__main__.QGuiApplication.styleHints",
          return_value=_mock_style_hints(Qt.ColorScheme.Light),
      ):
          _apply_windows_palette(qapp)
      post = qapp.palette().color(QPalette.ColorRole.Window)
      # Early return on Light — unchanged
      assert post == pre


  def test_apply_windows_palette_unknown_scheme_noop(qapp):
      pre = qapp.palette().color(QPalette.ColorRole.Window)
      with patch(
          "musicstreamer.__main__.QGuiApplication.styleHints",
          return_value=_mock_style_hints(Qt.ColorScheme.Unknown),
      ):
          _apply_windows_palette(qapp)
      post = qapp.palette().color(QPalette.ColorRole.Window)
      assert post == pre
  ```

  Key points:
  - Use pytest-qt's `qapp` fixture (not `qtbot` — no widget construction needed).
  - Patch `musicstreamer.__main__.QGuiApplication.styleHints` — this requires that
    `_apply_windows_palette` imports `QGuiApplication` at module scope. If plan 36-01's
    `__main__.py` imports it inside the function only, move the import to module scope OR
    patch `PySide6.QtGui.QGuiApplication.styleHints` directly. Decide at implementation time
    based on what's actually in `__main__.py`.
  - Between dark and light tests, the `qapp` fixture has a shared palette — the `light_noop`
    test compares post against pre to avoid test-order sensitivity from the dark test having
    set a dark palette.
  - Do NOT mock `sys.platform` — the helper itself doesn't check platform, it's the caller
    (`_run_gui`) that gates. Testing `_run_gui` with a mocked platform is out of scope (it would
    require mocking `QApplication.setStyle`, `MainWindow`, and `exec()` — too fragile).
