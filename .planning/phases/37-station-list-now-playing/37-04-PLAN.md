"""Phase 37: MainWindow integration tests using a FakePlayer monkeypatch.

Verifies the end-to-end signal flow Player → NowPlayingPanel + ToastOverlay
without requiring a real GStreamer pipeline. See RESEARCH.md §6 (Code Examples)
for the FakePlayer test-double pattern.
"""
from __future__ import annotations

import pytest
from PySide6.QtCore import QObject, Signal, Qt
from PySide6.QtWidgets import QSplitter

from musicstreamer.models import Station, StationStream
# ensure resource registration
from musicstreamer.ui_qt import icons_rc  # noqa: F401


class FakePlayer(QObject):
    """Mirrors the Player signal surface — no GStreamer."""
    title_changed   = Signal(str)
    failover        = Signal(object)
    offline         = Signal(str)
    playback_error  = Signal(str)
    elapsed_updated = Signal(int)
    youtube_resolved = Signal(object)
    youtube_resolution_failed = Signal(str)

    def __init__(self, parent=None) -> None:
        super().__init__(parent)
        self.play_calls = []
        self.pause_calls = 0
        self.stop_calls = 0
        self.set_volume_calls = []

    def play(self, station, on_title=None, preferred_quality="",
             on_failover=None, on_offline=None) -> None:
        self.play_calls.append((station, preferred_quality))

    def pause(self) -> None:
        self.pause_calls += 1

    def stop(self) -> None:
        self.stop_calls += 1

    def set_volume(self, v: float) -> None:
        self.set_volume_calls.append(v)


@pytest.fixture
def main_window(qtbot, monkeypatch, tmp_path):
    """Construct MainWindow with FakePlayer monkeypatched in.

    Uses tmp_path-rooted data dir so the test never touches user data.
    """
    # Redirect data dir to tmp_path
    from musicstreamer import paths
    monkeypatch.setattr(paths, "_root_override", str(tmp_path), raising=False)

    # Monkeypatch Player class in main_window's import namespace
    from musicstreamer.ui_qt import main_window as mw_mod
    monkeypatch.setattr(mw_mod, "Player", FakePlayer)

    mw = mw_mod.MainWindow()
    qtbot.addWidget(mw)
    return mw


def _make_station(id: int = 1, name: str = "Test FM", provider: str = "TestProv") -> Station:
    return Station(
        id=id, name=name, provider_name=provider,
        station_art_path=None, last_played_at=None, icy_disabled=False,
        streams=[],
    )


def test_central_widget_is_splitter(main_window):
    cw = main_window.centralWidget()
    assert isinstance(cw, QSplitter)
    assert cw.orientation() == Qt.Horizontal
    assert cw.count() == 2


def test_splitter_initial_sizes(main_window):
    sizes = main_window.centralWidget().sizes()
    assert sizes == [360, 840]


def test_toast_overlay_parented_to_central_widget(main_window):
    assert main_window.toast_overlay.parent() is main_window.centralWidget()


def test_title_changed_updates_icy_label(main_window):
    main_window._player.title_changed.emit("Artist - Track")
    assert main_window.now_playing.icy_label.text() == "Artist - Track"


def test_elapsed_updated_formats_below_one_hour(main_window):
    main_window._player.elapsed_updated.emit(125)
    assert main_window.now_playing.elapsed_label.text() == "2:05"


def test_failover_none_shows_exhausted_toast(main_window, qtbot):
    main_window._player.failover.emit(None)
    assert main_window.toast_overlay.label.text() == "Stream exhausted"
    assert main_window.toast_overlay.isVisible()


def test_failover_with_stream_shows_trying_next_toast(main_window):
    fake_stream = StationStream(id=1, station_id=1, url="http://x", quality="hi", priority=1, label=None) if hasattr(StationStream, "__init__") else object()
    main_window._player.failover.emit(fake_stream)
    assert main_window.toast_overlay.label.text() == "Stream failed, trying next\u2026"


def test_offline_shows_channel_offline_toast(main_window):
    main_window._player.offline.emit("somechannel")
    assert main_window.toast_overlay.label.text() == "Channel offline"


def test_playback_error_short_message(main_window):

    assert main_window.toast_overlay.label.text() == "Playback error: connection refused"


def test_playback_error_long_message_truncates_to_80_with_ellipsis(main_window):

    assert text.startswith("Playback error: ")
    body = text[len("Playback error: "):]

    assert len(body) == 80
    assert body.endswith("\u2026")


def test_station_activated_plays_binds_and_toasts(main_window, monkeypatch):
    station = _make_station()
    update_calls = []

    assert main_window.toast_overlay.label.text() == "Connecting\u2026"
    assert main_window.now_playing._station is station
    assert update_calls == [station.id]


def test_idle_state_icy_label(main_window):
    assert main_window.now_playing.icy_label.text() == "No station playing"


def test_construct_destroy_cycle_no_lifetime_error(qtbot, monkeypatch, tmp_path):
    """QA-05 lifetime guard: construct/destroy MainWindow without RuntimeError."""

        mw.deleteLater()
        qtbot.wait(50)
    # Reaching here without RuntimeError is the implicit pass
