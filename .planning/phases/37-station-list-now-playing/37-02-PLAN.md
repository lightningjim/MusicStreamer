```python
def on_elapsed_updated(self, seconds: int) -> None:
    if seconds < 3600:
        self.elapsed_label.setText(f"{seconds // 60}:{seconds % 60:02d}")
    else:
        h = seconds // 3600
        m = (seconds % 3600) // 60
        s = seconds % 60
        self.elapsed_label.setText(f"{h}:{m:02d}:{s:02d}")
```

Cover art adapter (RESEARCH §5, D-19):
```python
self._station: Station | None = None
self._cover_fetch_token: int = 0
self._last_cover_icy: str | None = None
self.cover_art_ready.connect(self._on_cover_art_ready, Qt.ConnectionType.QueuedConnection)

def on_title_changed(self, title: str) -> None:
    self.icy_label.setText(title or "")
    if title and not is_junk_title(title) and self._station is not None:
        if title != self._last_cover_icy:
            self._last_cover_icy = title
            self._fetch_cover_art_async(title)

def _fetch_cover_art_async(self, icy_title: str) -> None:
    self._cover_fetch_token += 1
    token = self._cover_fetch_token
    def _cb(path_or_none):
        # Worker thread — emit only, no widget access
        self.cover_art_ready.emit(path_or_none or "")
    fetch_cover_art(icy_title, _cb)

def _on_cover_art_ready(self, path: str) -> None:
    if not path:
        self._show_station_logo_in_cover_slot()
        return
    self._set_cover_pixmap(path)

def _set_cover_pixmap(self, path: str) -> None:
    pix = QPixmap(path)
    if pix.isNull():
        self._show_station_logo_in_cover_slot()
        return
    scaled = pix.scaled(QSize(160, 160), Qt.KeepAspectRatio, Qt.SmoothTransformation)
    self.cover_label.setPixmap(scaled)
```
