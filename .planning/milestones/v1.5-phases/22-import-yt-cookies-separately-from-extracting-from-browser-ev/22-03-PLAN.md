4. Add module-level helper class `_GoogleLoginWindow(Gtk.Window)`:
   ```python
   class _GoogleLoginWindow(Gtk.Window):
       """Embedded browser window for Google sign-in. Captures YouTube cookies after login."""
       
       def __init__(self, on_cookies_ready):
           super().__init__(title="Sign in with Google")
           self.set_default_size(800, 600)
           self._callback = on_cookies_ready
           self._cookies_extracted = False
           
           # Use an ephemeral data manager so we don't pollute the default context
           data_mgr = WebKit2.WebsiteDataManager.new_ephemeral()
           ctx = WebKit2.WebContext()
           self._webview = WebKit2.WebView.new_with_context(ctx)
           
           # Configure the data manager
           settings = self._webview.get_settings()
           settings.set_enable_javascript(True)
           
           self._webview.connect("load-changed", self._on_load_changed)
           self.set_child(self._webview)
           
           # Navigate to Google sign-in for YouTube
           self._webview.load_uri("https://accounts.google.com/signin/v2/identifier?service=youtube&continue=https%3A%2F%2Fwww.youtube.com%2F")
           
           # Handle window close
           self.connect("close-request", self._on_close_request)
       
       def _on_load_changed(self, webview, event):
           if event != WebKit2.LoadEvent.FINISHED:
               return
           uri = webview.get_uri() or ""
           # Detect successful login — user has been redirected to YouTube
           if ("youtube.com" in uri and "accounts.google" not in uri) or "myaccount.google.com" in uri:
               if not self._cookies_extracted:
                   self._cookies_extracted = True
                   self._extract_cookies()
       
       def _extract_cookies(self):
           mgr = self._webview.get_website_data_manager().get_cookie_manager()
           mgr.get_all_cookies(None, self._on_got_cookies, None)
       
       def _on_got_cookies(self, mgr, res, user_data):
           try:
               cookies = mgr.get_all_cookies_finish(res)
               # Debug: print first cookie's dir() to verify API (remove after confirming)
               if cookies:
                   # Filter to YouTube/Google domains only
                   yt_cookies = [c for c in cookies if any(
                       d in (c.get_domain() or "") for d in [".youtube.com", ".google.com", "youtube.com", "google.com"]
                   )]
                   GLib.idle_add(self._deliver_cookies, yt_cookies)
               else:
                   GLib.idle_add(self._deliver_cookies, None)
           except Exception:
               GLib.idle_add(self._deliver_cookies, None)
       
       def _deliver_cookies(self, cookies):
           self._callback(cookies)
           self.close()
           return False  # Remove from idle
       
       def _on_close_request(self, win):
           if not self._cookies_extracted:
               self._callback(None)
           return False
   ```

5. Add the Netscape cookie writer as a module-level function:
   ```python
   def _write_netscape_cookies(cookies, path: str):
       """Write WebKit2 cookie objects to Netscape format file."""
       os.makedirs(os.path.dirname(path), exist_ok=True)
       lines = ["# Netscape HTTP Cookie File", ""]
       for c in cookies:
           domain = c.get_domain() or ""
           domain_dot = "TRUE" if domain.startswith(".") else "FALSE"
           path_val = c.get_path() or "/"
           secure = "TRUE" if c.get_secure() else "FALSE"
           # get_expires returns a GLib.DateTime or None
           expires = c.get_expires()
           if expires:
               expiry = int(expires.to_unix())
           else:
               expiry = 0
           name = c.get_name() or ""
           value = c.get_value() or ""
           lines.append(f"{domain}\t{domain_dot}\t{path_val}\t{secure}\t{expiry}\t{name}\t{value}")
       with open(path, "w") as f:
           f.write("\n".join(lines) + "\n")
   ```

IMPORTANT implementation notes:
- The WebKit2 Cookie object's `get_expires()` may return a `Soup.Date` or `GLib.DateTime` depending on the binding version. The implementation must handle both: check if the return has `to_unix()` (GLib.DateTime) or `to_time_t()` (Soup.Date). If neither works, fall back to 0.
- If `get_all_cookies_finish` does NOT exist as a method (per Assumption A1), try `get_cookies_finish` or inspect the CookieManager with `dir()` to find the correct finish method. Print a diagnostic message if the API is different than expected.
- The `_GoogleLoginWindow` must NOT be created in a background thread — it runs in the GTK main loop.
