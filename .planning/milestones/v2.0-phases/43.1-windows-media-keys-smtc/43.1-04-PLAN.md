```python
def publish_metadata(self, station, title, cover_pixmap):
    # D-06: deferred imports
    from winrt.windows.media import MediaPlaybackType  # noqa: PLC0415
    from winrt.windows.storage.streams import (  # noqa: PLC0415
        InMemoryRandomAccessStream, DataWriter,
        RandomAccessStreamReference,
    )

    self._station = station
    self._title = title or ""

    du = self._smtc.display_updater
    du.type = MediaPlaybackType.MUSIC  # D-08

    if station is None:
        du.music_properties.title = ""
        du.music_properties.artist = ""
        du.thumbnail = None
    else:
        du.music_properties.title = self._title or station.name
        du.music_properties.artist = station.name
        du.thumbnail = self._build_thumbnail_ref(
            cover_pixmap, station.id,
            InMemoryRandomAccessStream, DataWriter, RandomAccessStreamReference,
        )

    du.update()  # D-08: single batched update per publish_metadata
```

<!-- Thumbnail helper — isolated so tests can exercise it independently and assert the D-03 revised InMemoryRandomAccessStream path. -->

```python
def _build_thumbnail_ref(self, cover_pixmap, station_id,
                        InMemoryRandomAccessStream, DataWriter, RandomAccessStreamReference):
    """Return a RandomAccessStreamReference wrapping the cached PNG bytes, or None on failure.

    D-03 revised: file:// URIs are NOT accepted by create_from_uri
    (Pitfall #2 — valid schemes are http/https/ms-appx/ms-appdata only).
    """
    if cover_pixmap is None:
        return None

    path = write_cover_png(cover_pixmap, station_id)
    if not path:
        return None

    try:
        with open(path, "rb") as f:
            data = f.read()
        stream = InMemoryRandomAccessStream()
        writer = DataWriter(stream)
        writer.write_bytes(data)
        # Pitfall #3: store_async() can deadlock on Qt's STA main thread.
        # Synchronous alternative: writer.store_async() returns an IAsyncOperation;
        # pywinrt exposes .get() (blocking wait). If .get() deadlocks on the VM,
        # Plan 06 UAT will surface it and Plan 05 can pivot to asyncio.run().
        # See CONTEXT D-11 discretion: "Timing of display_updater.update() call".
        store_op = writer.store_async()
        if hasattr(store_op, "get"):
            store_op.get()  # blocks until store completes
        stream.seek(0)
        return RandomAccessStreamReference.create_from_stream(stream)
    except Exception as exc:
        _log.warning("SMTC thumbnail build failed for station %s: %s", station_id, exc)
        return None
```
