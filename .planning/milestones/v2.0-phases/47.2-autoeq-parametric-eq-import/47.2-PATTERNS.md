**Table-driven test style** (from `test_aa_import.py` lines 35-69):

```python
def test_parse_single_pk_band():
    text = "Preamp: -6.2 dB\nFilter 1: ON PK Fc 105 Hz Gain -3.5 dB Q 0.7"
    p = parse_autoeq(text)
    assert p.preamp_db == -6.2
    assert len(p.bands) == 1
    assert p.bands[0].filter_type == "PK"

def test_parse_rejects_zero_on_filters():
    with pytest.raises(ValueError, match="no ON filters"):
        parse_autoeq("Preamp: 0 dB\nFilter 1: OFF PK Fc 100 Hz Gain 0 dB Q 1.0")

def test_parse_shelf_tokens_lsc_hsc():   # C-1
    text = "Filter 1: ON LSC Fc 105 Hz Gain -3 dB Q 0.7\nFilter 2: ON HSC Fc 8000 Hz Gain 2 dB Q 0.7"
    p = parse_autoeq(text)
    assert [b.filter_type for b in p.bands] == ["LSC", "HSC"]

def test_serialize_round_trip():
    original = "Preamp: -6.2 dB\nFilter 1: ON PK Fc 105 Hz Gain -3.5 dB Q 0.7\n"
    reparsed = parse_autoeq(serialize_autoeq(parse_autoeq(original)))
    assert reparsed.preamp_db == -6.2
```
