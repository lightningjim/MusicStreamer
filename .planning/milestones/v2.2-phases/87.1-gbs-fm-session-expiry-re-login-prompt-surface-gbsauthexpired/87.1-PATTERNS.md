**Success detection — all four conditions** (accounts_dialog.py lines 584–610):
```python
def _on_finished(self, exit_code: int, exit_status: QProcess.ExitStatus) -> None:
    proc = self._login_proc
    self._login_proc = None  # WR-03: clear guard before any emission

    last_event = self._parse_oauth_stderr(proc)  # inline or extracted to subprocess_utils

    netscape_text: str = ""
    if proc is not None:
        try:
            # CRITICAL: do NOT .strip() — Netscape format preserves leading newlines
            netscape_text = proc.readAllStandardOutput().data().decode(
                "utf-8", errors="replace"
            )
        except Exception:
            netscape_text = ""

    from musicstreamer.gbs_api import _validate_gbs_cookies

    success_category = last_event is None or last_event.get("category") == "Success"
    if (
        exit_code == 0
        and netscape_text
        and success_category
        and _validate_gbs_cookies(netscape_text)
    ):
        cookies_path = paths.gbs_cookies_path()
        os.makedirs(os.path.dirname(cookies_path), exist_ok=True)
        with open(cookies_path, "w", encoding="utf-8") as fh:
            fh.write(netscape_text)
        os.chmod(cookies_path, 0o600)  # T-40-03: restrict permissions immediately
        self.relogin_succeeded.emit()
    else:
        self.relogin_failed.emit("Login was cancelled or failed")
```

**FailedToStart guard** (accounts_dialog.py lines 627–652 — `errorOccurred` fires INSTEAD of `finished` on FailedToStart):
```python
def _on_error(self, error: QProcess.ProcessError) -> None:
    _log.warning("GBS relogin subprocess errorOccurred (QProcess.ProcessError=%s).", error)
    if error == QProcess.ProcessError.FailedToStart:
        # WR-03: reset re-entrancy guard FIRST so re-entrant calls see None immediately
        self._login_proc = None
        self.relogin_failed.emit("Could not start login helper")
    # Crashed fires errorOccurred AND finished — _on_finished failure path handles it
```
