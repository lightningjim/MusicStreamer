```python
def fetch_cover_art(
    icy_string: str,
    callback: callable,
    source: str = "auto",  # "auto" | "itunes_only" | "mb_only"

    """Per-station cover-art dispatch (D-01..D-04).

    - source='itunes_only' (D-03): legacy path; never calls MB.
    - source='mb_only' (D-04): never calls iTunes (no side call for genre).
    - source='auto' (D-02): iTunes first; on miss, MB fallback.

    Junk-title gate (is_junk_title) runs FIRST regardless of source.
    For MB paths, the ' - ' D-07 split happens inside the dispatch — bare
    titles short-circuit to callback(None) when source='mb_only'; in 'auto'
    mode, a bare title still tries iTunes (it works for some streams),
    and on iTunes miss the bare title is dropped (no MB fallback per D-07).
    """
```
