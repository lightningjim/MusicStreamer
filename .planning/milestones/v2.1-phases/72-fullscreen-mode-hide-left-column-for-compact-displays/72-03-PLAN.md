def _on_compact_toggle(self, checked: bool) -> None:
    if checked:
        self._splitter_sizes_before_compact = self._splitter.sizes()  # BEFORE hide (Pitfall 1)
        self.station_panel.hide()
        self._install_peek_hover_filter()                              # Plan 04 fills body

            self._splitter_sizes_before_compact = None                 # Pitfall 5 reset
